 20  # take a list of files and (optionally) a list of paths 
 21  # return the first existing file found in the paths 
 22  # [file1, file2, file3], [path1, path2, path3] 
 23  # search order is: 
 24  # path1/file1 
 25  # path1/file2 
 26  # path1/file3 
 27  # path2/file1 
 28  # path2/file2 
 29  # path2/file3 
 30  # path3/file1 
 31  # path3/file2 
 32  # path3/file3 
 33   
 34  # first file found with os.path.exists() is returned 
 35  # no file matches raises ansibleerror 
 36  # EXAMPLES 
 37  #  - name: copy first existing file found to /some/file 
 38  #    action: copy src=$item dest=/some/file 
 39  #    with_first_found:  
 40  #     - files: foo ${inventory_hostname} bar 
 41  #       paths: /tmp/production /tmp/staging 
 42   
 43  # that will look for files in this order: 
 44  # /tmp/production/foo 
 45  #                 ${inventory_hostname} 
 46  #                 bar 
 47  # /tmp/staging/foo 
 48  #              ${inventory_hostname} 
 49  #              bar 
 50                     
 51  #  - name: copy first existing file found to /some/file 
 52  #    action: copy src=$item dest=/some/file 
 53  #    with_first_found:  
 54  #     - files: /some/place/foo ${inventory_hostname} /some/place/else 
 55   
 56  #  that will look for files in this order: 
 57  #  /some/place/foo 
 58  #  $relative_path/${inventory_hostname} 
 59  #  /some/place/else 
 60   
 61  # example - including tasks: 
 62  #  tasks: 
 63  #  - include: $item 
 64  #    with_first_found: 
 65  #     - files: generic 
 66  #       paths: tasks/staging tasks/production 
 67  # this will include the tasks in the file generic where it is found first (staging or production)

 77  # example skipping if no matched files 
 78  # First_found also offers the ability to control whether or not failing 
 79  # to find a file returns an error or not 
 80  # 
 81  #- name: first found file - or skip 
 82  #  action: copy src=$item dest=/etc/file.cfg 
 83  #  with_first_found: 
 84  #  - files: foo.${inventory_hostname} 
 85  #    skip: true 
 86   
 87  # example a role with default configuration and configuration per host 
 88  # you can set multiple terms with their own files and paths to look through. 
 89  # consider a role that sets some configuration per host falling back on a default config. 
 90  # 
 91  #- name: some configuration template 
 92  #  template: src={{ item }} dest=/etc/file.cfg mode=0444 owner=root group=root 
 93  #  with_first_found: 
 94  #   - files: 
 95  #      - ${inventory_hostname}/etc/file.cfg 
 96  #     paths: 
 97  #      - ../../../templates.overwrites 
 98  #      - ../../../templates 
 99  #   - files: 
100  #      - etc/file.cfg 
101  #     paths: 
102  #      - templates 
103   
104  # the above will return an empty list if the files cannot be found at all 
105  # if skip is unspecificed or if it is set to false then it will return a list  
106  # error which can be caught bye ignore_errors: true for that action. 
107   
108  # finally - if you want you can use it, in place to replace first_available_file: 
109  # you simply cannot use the - files, path or skip options. simply replace 
110  # first_available_file with with_first_found and leave the file listing in place 
111  # 
112  #
