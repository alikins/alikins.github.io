 30      ''' 
 31      This is intended for allowing things like: 
 32      with_items: a_list_variable 
 33   
 34      Where Jinja2 would return a string but we do not want to allow it to 
 35      call functions (outside of Jinja2, where the env is constrained). If 
 36      the input data to this function came from an untrusted (remote) source, 
 37      it should first be run through _clean_data_struct() to ensure the data 
 38      is further sanitized prior to evaluation. 
 39   
 40      Based on: 
 41      http://stackoverflow.com/questions/12523516/using-ast-and-whitelists-to-make-pythons-eval-safe 
 42      ''' 
 43   
 44      # define certain JSON types 
 45      # eg. JSON booleans are unknown to python eval() 
 46      JSON_TYPES = { 
 47          'false': False, 
 48          'null': None, 
 49          'true': True, 
 50      } 
 51   
 52      # this is the whitelist of AST nodes we are going to 
 53      # allow in the evaluation. Any node type other than 
 54      # those listed here will raise an exception in our custom 
 55      # visitor class defined below. 
 56      SAFE_NODES = set( 
 57          ( 
 58              ast.Add, 
 59              ast.BinOp, 
 60              ast.Call, 
 61              ast.Compare, 
 62              ast.Dict, 
 63              ast.Div, 
 64              ast.Expression, 
 65              ast.List, 
 66              ast.Load, 
 67              ast.Mult, 
 68              ast.Num, 
 69              ast.Name, 
 70              ast.Str, 
 71              ast.Sub, 
 72              ast.Tuple, 
 73              ast.UnaryOp, 
 74          ) 
 75      ) 
 76   
 77      # AST node types were expanded after 2.6 
 78      if sys.version_info[:2] >= (2, 7): 
 79          SAFE_NODES.update( 
 80              set( 
 81                  (ast.Set,) 
 82              ) 
 83          ) 
 84   
 85      # And in Python 3.4 too 
 86      if sys.version_info[:2] >= (3, 4): 
 87          SAFE_NODES.update( 
 88              set( 
 89                  (ast.NameConstant,) 
 90              ) 
 91          ) 
 92   
 93      filter_list = [] 
 94      for filter in filter_loader.all(): 
 95          filter_list.extend(filter.filters().keys()) 
 96   
 97      test_list = [] 
 98      for test in test_loader.all(): 
 99          test_list.extend(test.tests().keys()) 
100   
101      CALL_WHITELIST = C.DEFAULT_CALLABLE_WHITELIST + filter_list + test_list 
102   
103      class CleansingNodeVisitor(ast.NodeVisitor): 
104          def generic_visit(self, node, inside_call=False): 
105              if type(node) not in SAFE_NODES: 
106                  raise Exception("invalid expression (%s)" % expr) 
107              elif isinstance(node, ast.Call): 
108                  inside_call = True 
109              elif isinstance(node, ast.Name) and inside_call: 
110                  if hasattr(builtins, node.id) and node.id not in CALL_WHITELIST: 
111                      raise Exception("invalid function: %s" % node.id) 
112              # iterate over all child nodes 
113              for child_node in ast.iter_child_nodes(node): 
114                  self.generic_visit(child_node, inside_call) 

115   
116      if not isinstance(expr, string_types): 
117          # already templated to a datastructure, perhaps? 
118          if include_exceptions: 
119              return (expr, None) 
120          return expr 
121   
122      cnv = CleansingNodeVisitor() 
123      try: 
124          parsed_tree = ast.parse(expr, mode='eval') 
125          cnv.visit(parsed_tree) 
126          compiled = compile(parsed_tree, expr, 'eval') 
127          result = eval(compiled, JSON_TYPES, dict(locals)) 
128   
129          if include_exceptions: 
130              return (result, None) 
131          else: 
132              return result 
133      except SyntaxError as e: 
134          # special handling for syntax errors, we just return 
135          # the expression string back as-is 
136          if include_exceptions: 
137              return (expr, None) 
138          return expr 
139      except Exception as e: 
140          if include_exceptions: 
141              return (expr, e) 
142          return expr
