55      """Test whether a path is a mount point 
56      clone of os.path.ismount (from cpython Lib/posixpath.py) 
57      fixed to solve https://github.com/ansible/ansible-modules-core/issues/2186 
58      and workaround non-fixed http://bugs.python.org/issue2466 
59      this should be rewritten as soon as python issue 2466 is fixed 
60      probably check for python version and use os.path.ismount if fixed 
61   
62      to remove replace in this file ismount( -> os.path.ismount( and remove this 
63      function""" 
64   
65      try: 
66          s1 = os.lstat(path) 
67      except OSError: 
68          # the OSError should be handled with more care 
69          # it could be a "permission denied" but path is still a mount 
70          return False 
71      else: 
72          # A symlink can never be a mount point 
73          if os.path.stat.S_ISLNK(s1.st_mode): 
74              return False 
75   
76      parent = os.path.join(path, os.path.pardir) 
77      parent = os.path.realpath(parent) 
78   
79      try: 
80          s2 = os.lstat(parent) 
81      except OSError: 
82          # one should handle the returned OSError with more care to figure 
83          # out whether this is still a mount 
84          return False 
85   
86      if s1.st_dev != s2.st_dev: 
87          return True     # path/.. on a different device as path 
88      if s1.st_ino == s2.st_ino: 
89          return True     # path/.. is the same i-node as path, i.e. path=='/' 
90      return False
