 59 -def axapi_call(module, url, post=None): 

 60      ''' 
 61      Returns a datastructure based on the result of the API call 
 62      ''' 
 63      rsp, info = fetch_url(module, url, data=post) 
 64      if not rsp or info['status'] >= 400: 
 65          module.fail_json(msg="failed to connect (status code %s), error was %s" % (info['status'], info.get('msg', 'no error given'))) 
 66      try: 
 67          raw_data = rsp.read() 
 68          data = json.loads(raw_data) 
 69      except ValueError: 
 70          # at least one API call (system.action.write_config) returns 
 71          # XML even when JSON is requested, so do some minimal handling 
 72          # here to prevent failing even when the call succeeded 
 73          if 'status="ok"' in raw_data.lower(): 
 74              data = {"response": {"status": "OK"}} 
 75          else: 
 76              data = {"response": {"status": "fail", "err": {"msg": raw_data}}} 
 77      except: 
 78          module.fail_json(msg="could not read the result from the host") 
 79      finally: 
 80          rsp.close() 
 81      return data
