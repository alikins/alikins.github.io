 69      """Double backslashes within jinja2 expressions 
 70   
 71      A user may enter something like this in a playbook:: 
 72   
 73        debug: 
 74          msg: "Test Case 1\\3; {{ test1_name | regex_replace('^(.*)_name$', '\\1')}}" 
 75   
 76      The string inside of the {{ gets interpreted multiple times First by yaml. 
 77      Then by python.  And finally by jinja2 as part of it's variable.  Because 
 78      it is processed by both python and jinja2, the backslash escaped 
 79      characters get unescaped twice.  This means that we'd normally have to use 
 80      four backslashes to escape that.  This is painful for playbook authors as 
 81      they have to remember different rules for inside vs outside of a jinja2 
 82      expression (The backslashes outside of the "{{ }}" only get processed by 
 83      yaml and python.  So they only need to be escaped once).  The following 
 84      code fixes this by automatically performing the extra quoting of 
 85      backslashes inside of a jinja2 expression. 
 86   
 87      """ 
 88      if '\\' in data and '{{' in data: 
 89          new_data = [] 
 90          d2 = jinja_env.preprocess(data) 
 91          in_var = False 
 92   
 93          for token in jinja_env.lex(d2): 
 94              if token[1] == 'variable_begin': 
 95                  in_var = True 
 96                  new_data.append(token[2]) 
 97              elif token[1] == 'variable_end': 
 98                  in_var = False 
 99                  new_data.append(token[2]) 
100              elif in_var and token[1] == 'string': 
101                  # Double backslashes only if we're inside of a jinja2 variable 
102                  new_data.append(token[2].replace('\\','\\\\')) 
103              else: 
104                  new_data.append(token[2]) 
105   
106          data = ''.join(new_data) 
107   
108      return data

110 -def _count_newlines_from_end(in_str): 

111      ''' 
112      Counts the number of newlines at the end of a string. This is used during 
113      the jinja2 templating to ensure the count matches the input, since some newlines 
114      may be thrown away during the templating. 
115      ''' 
116   
117      try: 
118          i = len(in_str) 
119          j = i -1 
120          while in_str[j] == '\n': 
121              j -= 1 
122          return i - 1 - j 
123      except IndexError: 
124          # Uncommon cases: zero length string and string containing only newlines 
125          return i
