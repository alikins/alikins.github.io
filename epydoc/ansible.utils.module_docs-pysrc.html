 47      """ 
 48      Search for assignment of the DOCUMENTATION and EXAMPLES variables 
 49      in the given file. 
 50      Parse DOCUMENTATION from YAML and return the YAML doc or None 
 51      together with EXAMPLES, as plain text. 
 52   
 53      DOCUMENTATION can be extended using documentation fragments 
 54      loaded by the PluginLoader from the module_docs_fragments 
 55      directory. 
 56      """ 
 57   
 58      doc = None 
 59      plainexamples = None 
 60      returndocs = None 
 61   
 62      try: 
 63          # Thank you, Habbie, for this bit of code :-) 
 64          M = ast.parse(''.join(open(filename))) 
 65          for child in M.body: 
 66              if isinstance(child, ast.Assign): 
 67                  for t in child.targets: 
 68                      try: 
 69                          theid = t.id 
 70                      except AttributeError as e: 
 71                          # skip errors can happen when trying to use the normal code 
 72                          display.warning("Failed to assign id for %s on %s, skipping" % (t, filename)) 
 73                          continue 
 74   
 75                      if 'DOCUMENTATION' in theid: 
 76                          doc = AnsibleLoader(child.value.s, file_name=filename).get_single_data() 
 77                          fragments = doc.get('extends_documentation_fragment', []) 
 78   
 79                          if isinstance(fragments, basestring): 
 80                              fragments = [ fragments ] 
 81   
 82                          # Allow the module to specify a var other than DOCUMENTATION 
 83                          # to pull the fragment from, using dot notation as a separator 
 84                          for fragment_slug in fragments: 
 85                              fragment_slug = fragment_slug.lower() 
 86                              if '.' in fragment_slug: 
 87                                  fragment_name, fragment_var = fragment_slug.split('.', 1) 
 88                                  fragment_var = fragment_var.upper() 
 89                              else: 
 90                                  fragment_name, fragment_var = fragment_slug, 'DOCUMENTATION' 
 91   
 92                              fragment_class = fragment_loader.get(fragment_name) 
 93                              assert fragment_class is not None 
 94   
 95                              fragment_yaml = getattr(fragment_class, fragment_var, '{}') 
 96                              fragment = AnsibleLoader(fragment_yaml, file_name=filename).get_single_data() 
 97   
 98                              if fragment.has_key('notes'): 
 99                                  notes = fragment.pop('notes') 
100                                  if notes: 
101                                      if not doc.has_key('notes'): 
102                                          doc['notes'] = [] 
103                                      doc['notes'].extend(notes) 
104   
105                              if 'options' not in fragment.keys(): 
106                                  raise Exception("missing options in fragment, possibly misformatted?") 
107   
108                              for key, value in fragment.items(): 
109                                  if not doc.has_key(key): 
110                                      doc[key] = value 
111                                  else: 
112                                      if isinstance(doc[key], MutableMapping): 
113                                          doc[key].update(value) 
114                                      elif isinstance(doc[key], MutableSet): 
115                                          doc[key].add(value) 
116                                      elif isinstance(doc[key], MutableSequence): 
117                                          doc[key] = sorted(frozenset(doc[key] + value))

126      except: 
127          display.error("unable to parse %s" % filename) 
128          if verbose == True:
