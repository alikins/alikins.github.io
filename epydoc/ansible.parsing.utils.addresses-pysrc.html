 72  # A hostname label, e.g. 'foo' in 'foo.example.com'. Consists of alphanumeric 
 73  # characters plus dashes (and underscores) or valid ranges. The label may not 
 74  # start or end with a hyphen or an underscore. This is interpolated into the 
 75  # hostname pattern below. We don't try to enforce the 63-char length limit.

 77  label = r''' 
 78      (?:[\w]|{range})                    # Starts with an alphanumeric or a range 
 79      (?:[\w_-]|{range})*                 # Then zero or more of the same or [_-] 
 80      (?<![_-])                           # ...as long as it didn't end with [_-]

106              $

119      # This matches an IPv6 address, but also permits range expressions. 
120      # 
121      # This expression looks complex, but it really only spells out the various 
122      # combinations in which the basic unit of an IPv6 address (0..ffff) can be 
123      # written, from :: to 1:2:3:4:5:6:7:8, plus the IPv4-in-IPv6 variants such 
124      # as ::ffff:192.0.2.3. 
125      # 
126      # Note that we can't just use ipaddress.ip_address() because we also have to 
127      # accept ranges in place of each component. 
128   
129      'ipv6': re.compile( 
130          r'''^ 
131              (?:{0}:){{7}}{0}|           # uncompressed: 1:2:3:4:5:6:7:8 
132              (?:{0}:){{1,6}}:|           # compressed variants, which are all 
133              (?:{0}:)(?::{0}){{1,6}}|    # a::b for various lengths of a,b 
134              (?:{0}:){{2}}(?::{0}){{1,5}}| 
135              (?:{0}:){{3}}(?::{0}){{1,4}}| 
136              (?:{0}:){{4}}(?::{0}){{1,3}}| 
137              (?:{0}:){{5}}(?::{0}){{1,2}}| 
138              (?:{0}:){{6}}(?::{0})|      # ...all with 2 <= a+b <= 7 
139              :(?::{0}){{1,6}}|           # ::ffff(:ffff...) 
140              {0}?::|                     # ffff::, :: 
141                                          # ipv4-in-ipv6 variants 
142              (?:0:){{6}}(?:{0}\.){{3}}{0}| 
143              ::(?:ffff:)?(?:{0}\.){{3}}{0}| 
144              (?:0:){{5}}ffff:(?:{0}\.){{3}}{0} 
145              $ 
146          '''.format(ipv6_component), re.X|re.I 
147      ),

149      # This matches a hostname or host pattern including [x:y(:z)] ranges. 
150      # 
151      # We roughly follow DNS rules here, but also allow ranges (and underscores). 
152      # In the past, no systematic rules were enforced about inventory hostnames, 
153      # but the parsing context (e.g. shlex.split(), fnmatch.fnmatch()) excluded 
154      # various metacharacters anyway.

169 -def parse_address(address, allow_ranges=False): 

170      """ 
171      Takes a string and returns a (host, port) tuple. If the host is None, then 
172      the string could not be parsed as a host identifier with an optional port 
173      specification. If the port is None, then no port was specified. 
174   
175      The host identifier may be a hostname (qualified or not), an IPv4 address, 
176      or an IPv6 address. If allow_ranges is True, then any of those may contain 
177      [x:y] range specifications, e.g. foo[1:3] or foo[0:5]-bar[x-z]. 
178   
179      The port number is an optional :NN suffix on an IPv4 address or host name, 
180      or a mandatory :NN suffix on any square-bracketed expression: IPv6 address, 
181      IPv4 address, or host name. (This means the only way to specify a port for 
182      an IPv6 address is to enclose it in square brackets.) 
183      """ 
184   
185      # First, we extract the port number if one is specified. 
186   
187      port = None 
188      for matching in ['bracketed_hostport', 'hostport']: 
189          m = patterns[matching].match(address) 
190          if m: 
191              (address, port) = m.groups() 
192              port = int(port) 
193              continue 
194   
195      # What we're left with now must be an IPv4 or IPv6 address, possibly with 
196      # numeric ranges, or a hostname with alphanumeric ranges. 
197   
198      host = None 
199      for matching in ['ipv4', 'ipv6', 'hostname']: 
200          m = patterns[matching].match(address) 
201          if m: 
202              host = address 
203              continue 
204   
205      # If it isn't any of the above, we don't understand it. 
206      if not host: 
207          raise AnsibleError("Not a valid network hostname: %s" % address) 
208   
209      # If we get to this point, we know that any included ranges are valid. 
210      # If the caller is prepared to handle them, all is well. 
211      # Otherwise we treat it as a parse failure. 
212      if not allow_ranges and '[' in host: 
213          raise AnsibleParserError("Detected range in host but was asked to ignore ranges") 
214   
215      return (host, port)
