31      cmdargs = shlex.split(cmd) 
32      p = subprocess.Popen(cmdargs, stdout=subprocess.PIPE, stderr=subprocess.PIPE) 
33   
34      stdout = '' 
35      stderr = '' 
36      rpipes = [p.stdout, p.stderr] 
37      while True: 
38          rfd, wfd, efd = select.select(rpipes, [], rpipes, 1) 
39   
40          if p.stdout in rfd: 
41              dat = os.read(p.stdout.fileno(), readsize) 
42              if live: 
43                  sys.stdout.write(dat) 
44              stdout += dat 
45              if dat == '': 
46                  rpipes.remove(p.stdout) 
47          if p.stderr in rfd: 
48              dat = os.read(p.stderr.fileno(), readsize) 
49              stderr += dat 
50              if live: 
51                  sys.stdout.write(dat) 
52              if dat == '': 
53                  rpipes.remove(p.stderr) 
54          # only break out if we've emptied the pipes, or there is nothing to 
55          # read from and the process has finished. 
56          if (not rpipes or not rfd) and p.poll() is not None: 
57              break 
58          # Calling wait while there are still pipes to read can cause a lock 
59          elif not rpipes and p.poll() == None: 
60              p.wait() 
61   
62      return p.returncode, stdout, stderr
