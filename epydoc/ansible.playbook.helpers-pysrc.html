 36      ''' 
 37      Given a list of mixed task/block data (parsed from YAML), 
 38      return a list of Block() objects, where implicit blocks 
 39      are created for each bare Task. 
 40      ''' 
 41   
 42      # we import here to prevent a circular dependency with imports 
 43      from ansible.playbook.block import Block 
 44   
 45      assert isinstance(ds, (list, type(None))) 
 46   
 47      block_list = [] 
 48      if ds: 
 49          for block in ds: 
 50              b = Block.load( 
 51                  block, 
 52                  play=play, 
 53                  parent_block=parent_block, 
 54                  role=role, 
 55                  task_include=task_include, 
 56                  use_handlers=use_handlers, 
 57                  variable_manager=variable_manager, 
 58                  loader=loader 
 59              ) 
 60              # Implicit blocks are created by bare tasks listed in a play without 
 61              # an explicit block statement. If we have two implicit blocks in a row, 
 62              # squash them down to a single block to save processing time later. 
 63              if b._implicit and len(block_list) > 0 and block_list[-1]._implicit: 
 64                  for t in b.block: 
 65                      t._block = block_list[-1] 
 66                  block_list[-1].block.extend(b.block) 
 67              else: 
 68                  block_list.append(b) 
 69   
 70      return block_list
