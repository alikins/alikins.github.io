19  from __future__ import (absolute_import, division, print_function)

22  ''' 
23  Compat six library.  RHEL7 has python-six 1.3.0 which is too old 
24  ''' 
25  # The following makes it easier for us to script updates of the bundled code 
26  _BUNDLED_METADATA = { "pypi_name": "six", "version": "1.10.0" } 
27   
28  import os.path 
29   
30  try: 
31      import six as _system_six 
32  except ImportError: 
33      _system_six = None 
34   
35  if _system_six: 
36      # If we need some things from even newer versions of six, then we need to 
37      # use our bundled copy instead 
38   
39         if ( # Added in six-1.8.0 
40              not hasattr(_system_six.moves, 'shlex_quote') or 
41              # Added in six-1.4.0 
42              not hasattr(_system_six, 'byte2int') or 
43              not hasattr(_system_six, 'add_metaclass') or 
44              not hasattr(_system_six.moves, 'urllib') 
45              ): 
46   
47          _system_six = False 
48   
49  if _system_six: 
50      six = _system_six 
51  else: 
52      from . import _six as six 
53  six_py_file = '{0}.py'.format(os.path.splitext(six.__file__)[0]) 
54  exec(open(six_py_file, 'rb').read())
