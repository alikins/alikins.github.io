 23  options: 
 24    cloud: 
 25      description: 
 26        - Named cloud to operate against. Provides default values for I(auth) and 
 27          I(auth_type). This parameter is not needed if I(auth) is provided or if 
 28          OpenStack OS_* environment variables are present. 
 29      required: false 
 30    auth: 
 31      description: 
 32        - Dictionary containing auth information as needed by the cloud's auth 
 33          plugin strategy. For the default I(password) plugin, this would contain 
 34          I(auth_url), I(username), I(password), I(project_name) and any 
 35          information about domains if the cloud supports them. For other plugins, 
 36          this param will need to contain whatever parameters that auth plugin 
 37          requires. This parameter is not needed if a named cloud is provided or 
 38          OpenStack OS_* environment variables are present. 
 39      required: false 
 40    auth_type: 
 41      description: 
 42        - Name of the auth plugin to use. If the cloud uses something other than 
 43          password authentication, the name of the plugin should be indicated here 
 44          and the contents of the I(auth) parameter should be updated accordingly. 
 45      required: false 
 46      default: password 
 47    region_name: 
 48      description: 
 49        - Name of the region. 
 50      required: false 
 51    availability_zone: 
 52      description: 
 53        - Name of the availability zone. 
 54      required: false 
 55    wait: 
 56      description: 
 57        - Should ansible wait until the requested resource is complete. 
 58      required: false 
 59      default: "yes" 
 60      choices: ["yes", "no"] 
 61    timeout:

 63        - How long should ansible wait for the requested resource. 
 64      required: false 
 65      default: 180 
 66    api_timeout: 
 67      description: 
 68        - How long should the socket layer wait before timing out for API calls.

 71      default: None

 92      required: false 
 93      default: None
